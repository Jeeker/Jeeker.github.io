$.cookie = function(key, value, options) {

    // key and at least value given, set cookie...
    if (arguifments.length > 1 && (!/Object/.test(Object.prototype.toString.call(value)) || value === null || value === undefined)) {
        options = $.extend({}, options);

        if (value === null || value === undefined) {
            options.expires = -1;
        }

        if (typeof options.expires === 'number') {
            var days = options.expires, t = options.expires = new Date();
            t.setDate(t.getDate() + days);
        }

        value = String(value);

        return (document.cookie = [
            encodeURIComponent(key), '=', options.raw ? value : encodeURIComponent(value),
            options.expires ? '; expires=' + options.expires.toUTCString() : '', // use expires attribute, max-age is not supported by IE
            options.path    ? '; path=' + options.path : '',
            options.domain  ? '; domain=' + options.domain : '',
            options.secure  ? '; secure' : ''
        ].join(''));
    }

    // key and possibly options given, get cookie...
    options = value || {};
    var decode = options.raw ? function(s) { return s; } : decodeURIComponent;

    var pairs = document.cookie.split('; ');
    for (var i = 0, pair; pair = pairs[i] && pairs[i].split('='); i++) {
        if (decode(pair[0]) === key) return decode(pair[1] || ''); // IE saves cookies with empty string as "c; ", e.g. without "=" as opposed to EOMB, thus pair[1] may be undefined
    }
    return null;
};

class PageNav(object):
    def __init__(self, articles, num_per_page, first_page_name, other_page_name = 'page-%d'):
        assert isinstance(articles, list) and len(articles) > 0 and num_per_page > 0
        self.articles = articles
        self.num_per_page = num_per_page
        self.first_page_name = first_page_name
        self.other_page_name = other_page_name

    # 页面数
    def getPageCount(self):
        count = len(self.articles) / self.num_per_page;
        if len(self.articles) % self.num_per_page > 0:
            count += 1
        return count

    # 获取特定页面的数据 [articles, deploy_file_name]
    def getDataByPageNum(self, num):
        assert 0 < num <= self.getPageCount()
        start_index = self.num_per_page * (num-1)
        end_index = start_index + self.num_per_page + 1
        deploy_file_name = self.first_page_name;
        if num > 1:
            deploy_file_name = self.other_page_name % num
        return [self.articles[start_index:end_index], deploy_file_name]

03 Dec

　　Mac下系统原生自带了全系统搜索工具Spotlight

http://sina.com/

快速启动应用

　　这是Alfred最基本的应用了，基本两个字母就能准确定位到想要的App了，勾选Fetures - Applications - Partail matching for Applications还能进行模糊匹配。

搜索文件

　　Alfred提供了诸如find open in等关键词用于搜索文件，不过个人感觉比系统自带的Spotlight还是稍逊一些，尤其是在Lion和Mountain Lion中均对Spotlight的功能做了大量的加强，所以如果是要搜索文件的话还是直接用Spotlight更加快速高效。

快速遍历文件系统

　　虽然Alfred的文件搜索功能稍弱，但遍历文件系统却是无比的方便， 在Alfred搜索框输入/（定位到系统根目录）或~（定位到当前用户的家目录）即可使用键盘快速遍历文件系统，遍历到所需文件后ENTER(若是文件夹则使用CMD+ O)可快速打开目标，CMD + ENTER可在Finder中定位目标，→可执行一些常用命令如拷贝、移动等。

　　快捷键CMD + OPTION + /或previous也可激活文件导航，同时它还会将文件路径定位到最近一次浏览的位置，CMD + OPTION + \或action则可激活当前Finder中选中项的常用命令操作。

控制iTunes

　　itunes或CMD + OPTION + ENTER可启动iTunes Mini Player，进行iTunes控制或媒体库搜索、遍历等操作，同时使用play next random等关键词命令也可对iTunes进行相应的快速控制。

自定义搜索

　　Alfred自带了大量网络搜索引擎，可在Features - Web Search中根据个人需要启用或禁用，当然还可以自定义搜索，如搜索淘宝http://s.taobao.com/search?q={query}，配合关键词tb就可以在Alfred的搜索框中直接搜索淘宝了。

  • Advanced - Keyboard - Force Keyboard 选择U.S. 无论何时进入Alfred搜索框都将使用英文键盘，避免你正在使用中文输入法时进入Alfred搜索框需进行额外的输入法切换操作

一个超简单的在Google中直接访问目标网址的脚本

22 Nov

　　前段时间因开某会，悲催的Google一如以往的无法正常使用，常常搜索结果页面顺利出现了，点击链接却无法打开或打开很慢，产生这种现象的原因是Google通过跳转后才把目标网址返回给你，于是领导一个不高兴就又被墙了。

　　下面的这段很简单的代码就是让你避免使用Google跳转，直接访问目标网址。

(function(){
    function getRealUrl(l) {
        if (l.indexOf('/url?') < 0)
            return null;
        var a = document.createElement('a'); 
        a.href = l;
        seg = a.search.replace(/^\?/,'').split('&');
        for (i = 0; i < seg.length; i++) { 
            if (!seg[i])
                continue;
            s = seg[i].split('='); 
            if (s[0] == 'url')
                return decodeURIComponent( s[1] );
        } 
        return null;
    }

    /**
     * 直接在载入http:://www.google.com/url?.... 中重定向到真实的目标网址
     * 问题：当目标网址无法访问时，地址栏中的地址信息并未改变到目标网址，可能产生误解
     */
    var real = getRealUrl(location.href);
    if (real) {
        window.location.href = real;
        return;
    };

    document.addEventListener('click', function(e){
        for(a = e.target; a; a = a.parentNode) {
            if(a.localName != 'a') 
                continue;
            real = getRealUrl( a.getAttribute('href') );
            if (real) {
                a.setAttribute( 'href', real );
                a.removeAttribute('onmousedown');
            }
            break;
        }
    }, false);
})();
